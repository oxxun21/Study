arguments

함수 안에서 함수의 여러가지 정보(특히 인자)를 담고 있음

        function a(arg){ }  > arg : 매개변수 parameter
        a(1); > 1은 인자 arguments

arguments.length : 함수에 정의된 인자의 수를 알 수 있음

apply

        function sum(arg1, arg2){
            return arg1+arg2;
        }
        alert(sum.apply(null, [1,2])) == sum(1,2)
        // 3
        // null말고 다른 걸 넣을 때 쓰라고 있는거
    
        o1 = {val1:1, val2:2, val3:3}
        o2 = {v1:10, v2:50, v3:100, v4:25}
        function sum(){
            var _sum = 0;
                for(name in this){
                    _sum += this[name];
                }
            return _sum;
        }
        alert(sum.apply(o1))    // 6
        alert(sum.apply(o2))    // 185

        - apply를 사용 안했을 때 -
        function sum(){
            var _sum = 0;
                for(name in this){
                    if(typeof this[name] !== 'function')
                        _sum += this[name];
                }
            return _sum;
        }
        o1 = {val1:1, val2:2, val3:3, sum:sum}
        o2 = {v1:10, v2:50, v3:100, v4:25, sum:sum}
        alert(o1.sum());    // 6
        alert(o2.sum());    // 185
