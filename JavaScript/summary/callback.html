값으로서의 함수

자바스크립트에서는 함수도 객체, 일종의 값 : 다른 언어 함수와 다른 점은 함수 자체가 값이 될 수 있는 점이다.

        var a = function () {}
    
        a = {
            b : function(){}
        };

        > a는 함수를 값으로 가지고 있음, a{속성}; 
        > b : 변수, key의 역할
        > function : value, method(메소드, 객체 안에 정의되어 있는 함수)

함수는 변수, 매개 변수, 리턴 값, 배열의 값으로도 사용할 수 있음 : first-class citizen(object)

        리턴 값으로 사용 )

        function cal(mode){
            var funcs = {
                'plus' : function(left, right) {return left + right},
                'minus' : function(left, right) {return left - right}
            }
            return funcs[mode];
        }
        alert(cal('plus')(2,1));    // cal > funcs의 mode 값 > plus > plus 리턴, 호출 > 2,1 인자 대입
        alert(cal('minus')(2,1));

        배열의 값으로 사용 ) 

        var process = [
            function(input) {return input + 10;},   // 리턴 값 11
            function(input) {return input * input;},    // 리턴 값 121
            function(input) {return input / 2;}     // 리턴 값 60.5
        ];
        var input = 1;
        for(var i = 0; i < process.length; i++){    // length == 0,1,2
            input = process[i](input);      // i값이 0일때 process 첫번째 원소에 대입 > i값이 1이면 리턴 값이 11로 계산 시작
        }
        alert(input);
        // 마지막 리턴 값 60.5 출력
